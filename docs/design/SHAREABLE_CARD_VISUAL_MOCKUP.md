## Canvas 2D Implementation Code

### Full Rendering Function

```typescript
// shareCard.ts
import { createCanvas, registerFont } from 'canvas';

interface UserCardData {
  rank: number;
  score: number;
  avatarUrl: string;
  username: string;
  tier: 'S' | 'A' | 'B' | 'C';
}

async function generateSIGNALCard(user: UserCardData): Promise<Buffer> {
  const width = 1200;
  const height = 630;
  const canvas = createCanvas(width, height);
  const ctx = canvas.getContext('2d');

  // ============================================
  // 1. BACKGROUND
  // ============================================
  ctx.fillStyle = '#0A0A0F';
  ctx.fillRect(0, 0, width, height);

  // ============================================
  // 2. HEADER: "⭐ SIGNAL"
  // ============================================
  ctx.font = 'bold 32px Inter';
  ctx.fillStyle = '#F59E0B'; // Gold
  ctx.textAlign = 'left';
  ctx.textBaseline = 'top';
  ctx.fillText('⭐ SIGNAL', 40, 30);

  // ============================================
  // 3. AVATAR SECTION (with glow & shine)
  // ============================================
  const avatarX = width / 2;
  const avatarY = 240;
  const avatarRadius = 100;

  // 3a. Draw radial glow (behind avatar)
  const glowGradient = ctx.createRadialGradient(
    avatarX, avatarY, 0,
    avatarX, avatarY, avatarRadius + 50
  );
  glowGradient.addColorStop(0, 'rgba(245, 158, 11, 0.3)');
  glowGradient.addColorStop(0.6, 'rgba(245, 158, 11, 0.1)');
  glowGradient.addColorStop(1, 'rgba(245, 158, 11, 0)');

  ctx.fillStyle = glowGradient;
  ctx.beginPath();
  ctx.arc(avatarX, avatarY, avatarRadius + 50, 0, Math.PI * 2);
  ctx.fill();

  // 3b. Draw avatar circle (placeholder or actual image)
  ctx.fillStyle = '#27272A';
  ctx.beginPath();
  ctx.arc(avatarX, avatarY, avatarRadius, 0, Math.PI * 2);
  ctx.fill();

  // Load and draw actual avatar image
  try {
    const avatarImg = await loadImage(user.avatarUrl);
    ctx.save();
    ctx.beginPath();
    ctx.arc(avatarX, avatarY, avatarRadius, 0, Math.PI * 2);
    ctx.clip();
    ctx.drawImage(
      avatarImg,
      avatarX - avatarRadius,
      avatarY - avatarRadius,
      avatarRadius * 2,
      avatarRadius * 2
    );
    ctx.restore();
  } catch (err) {
    console.warn('Failed to load avatar:', err);
    // Fallback: just show the glow with tier initial
    ctx.fillStyle = '#F59E0B';
    ctx.font = 'bold 48px Inter';
    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';
    ctx.fillText(user.tier, avatarX, avatarY);
  }

  // 3c. Draw holographic shine effect (45° overlay)
  const shineGradient = ctx.createLinearGradient(
    avatarX - 150, avatarY - 150,
    avatarX + 150, avatarY + 150
  );
  shineGradient.addColorStop(0, 'rgba(255, 255, 255, 0)');
  shineGradient.addColorStop(0.5, 'rgba(255, 255, 255, 0.15)');
  shineGradient.addColorStop(1, 'rgba(255, 255, 255, 0)');

  ctx.fillStyle = shineGradient;
  ctx.beginPath();
  ctx.arc(avatarX, avatarY, avatarRadius, 0, Math.PI * 2);
  ctx.fill();

  // 3d. Avatar border (subtle gold ring)
  ctx.strokeStyle = 'rgba(245, 158, 11, 0.5)';
  ctx.lineWidth = 3;
  ctx.beginPath();
  ctx.arc(avatarX, avatarY, avatarRadius, 0, Math.PI * 2);
  ctx.stroke();

  // ============================================
  // 4. DIVIDER LINE (above statement)
  // ============================================
  drawDashedLine(ctx, 100, 360, 1100, 360, '#27272A', 20, 15);

  // ============================================
  // 5. STATEMENT: "I CALLED IT" (HERO)
  // ============================================
  ctx.font = 'bold 56px "Plus Jakarta Sans"';
  ctx.fillStyle = '#FAFAFA';
  ctx.textAlign = 'center';
  ctx.textBaseline = 'middle';

  // Measure text to center properly
  const statementText = 'I CALLED IT';
  const metrics = ctx.measureText(statementText);

  ctx.fillText(statementText, width / 2, 430);

  // ============================================
  // 6. DIVIDER LINE (below statement)
  // ============================================
  drawDashedLine(ctx, 100, 500, 1100, 500, '#27272A', 20, 15);

  // ============================================
  // 7. STATS ROW (rank, score, tier)
  // ============================================
  ctx.font = '500 18px "JetBrains Mono"';
  ctx.fillStyle = '#A1A1AA';
  ctx.textAlign = 'center';
  ctx.letterSpacing = '2px';

  const statsText = `Rank #${user.rank}  |  ${user.score.toLocaleString()} Points  |  ★ Top Caller`;
  ctx.fillText(statsText, width / 2, 550);

  // ============================================
  // 8. FOOTER: "Generated by Foresight.fun"
  // ============================================
  ctx.font = '400 14px Inter';
  ctx.fillStyle = '#71717A';
  ctx.textAlign = 'center';
  ctx.textBaseline = 'bottom';
  ctx.fillText('Generated by Foresight.fun', width / 2, height - 20);

  // ============================================
  // 9. OPTIONAL: Rank badge (top right)
  // ============================================
  if (user.rank <= 10) {
    drawRankBadge(ctx, width - 80, 80, user.rank);
  }

  return canvas.toBuffer('image/png');
}

// ============================================
// Helper: Draw dashed line
// ============================================
function drawDashedLine(
  ctx: CanvasRenderingContext2D,
  x1: number,
  y1: number,
  x2: number,
  y2: number,
  color: string,
  dashLength: number,
  gapLength: number
) {
  ctx.strokeStyle = color;
  ctx.lineWidth = 2;
  ctx.setLineDash([dashLength, gapLength]);
  ctx.beginPath();
  ctx.moveTo(x1, y1);
  ctx.lineTo(x2, y2);
  ctx.stroke();
  ctx.setLineDash([]); // Reset
}

// ============================================
// Helper: Draw rank badge
// ============================================
function drawRankBadge(
  ctx: CanvasRenderingContext2D,
  x: number,
  y: number,
  rank: number
) {
  const radius = 45;

  // Background circle
  ctx.fillStyle = '#F59E0B';
  ctx.beginPath();
  ctx.arc(x, y, radius, 0, Math.PI * 2);
  ctx.fill();

  // Rank number
  ctx.font = 'bold 36px "JetBrains Mono"';
  ctx.fillStyle = '#0A0A0F';
  ctx.textAlign = 'center';
  ctx.textBaseline = 'middle';
  ctx.fillText(`#${rank}`, x, y);

  // Border
  ctx.strokeStyle = 'rgba(245, 158, 11, 0.7)';
  ctx.lineWidth = 2;
  ctx.beginPath();
  ctx.arc(x, y, radius, 0, Math.PI * 2);
  ctx.stroke();
}

export { generateSIGNALCard };
```

## Alternative Statements (for A/B Testing)

You can swap out "I CALLED IT" for any of these:

### Option 1: "The signal was clear."
- Slightly longer (more explanation)
- More aligned with product name
- Risk: Reads more passive

### Option 2: "I predicted it."
- More active voice
- Less confident than "I called it"
- Good for users #20+

### Option 3: "★ Verified"
- Ultra-minimal (icon + word)
- Good for top 10 users
- Feels more premium

### Option 4: "My call paid off."
- Acknowledges the risk/reward
- Slightly less cocky
- Good for mid-tier users (#10-50)

**Recommendation:** "I CALLED IT" is strongest for all tiers (1-100). If you want per-tier variation, test it post-launch.
